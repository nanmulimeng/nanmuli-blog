## еҹәдәҺ Spring Boot 3.5 + Vue 3 + йҳҝйҮҢдә‘ECSпјҲж— иҜ„и®әжЁЎеқ—пјү

## дёҖгҖҒйЎ№зӣ®е®ҡдҪҚ

### 1.1 ж ёеҝғе®ҡдҪҚ
- **дёӘдәәжҠҖжңҜеҚҡе®ў**пјҡи®°еҪ•жҠҖжңҜеӯҰд№ гҖҒеҲҶдә«жҠҖжңҜж–Үз« 
- **дёӘдәәеұ•зӨәзҪ‘з«ҷ**пјҡеұ•зӨәдёӘдәәжҠҖиғҪгҖҒйЎ№зӣ®з»ҸеҺҶ
- **жҠҖжңҜж—Ҙеҝ—**пјҡжҜҸж—ҘжҠҖжңҜз¬”и®°гҖҒеӯҰд№ жҖ»з»“

### 1.2 з”ЁжҲ·и§’иүІ
- **д»…жңүдёҖдёӘз®ЎзҗҶе‘ҳ**пјҡиҮӘе·ұ
- **и®ҝе®ў**пјҡеҸӘиғҪжөҸи§ҲпјҢдёҚиғҪдә’еҠЁпјҲж— иҜ„и®әпјү

### 1.3 ж ёеҝғеҠҹиғҪ
1. вң… ж–Үз« з®ЎзҗҶпјҲMarkdownзј–иҫ‘гҖҒеҸ‘еёғгҖҒеҲҶзұ»гҖҒж Үзӯҫпјү
2. вң… жҠҖжңҜж—Ҙеҝ—пјҲеҝ«йҖҹи®°еҪ•гҖҒж—¶й—ҙзәҝеұ•зӨәпјү
3. вң… дёӘдәәеұ•зӨәпјҲе…ідәҺйЎөйқўгҖҒжҠҖиғҪеұ•зӨәгҖҒйЎ№зӣ®еұ•зӨәпјү
4. вң… еҶ…е®№жҗңзҙўпјҲе…Ёж–Үжҗңзҙўпјү
5. вң… AIиҫ…еҠ©пјҲжҷәиғҪж ҮзӯҫгҖҒж–Үз« ж‘ҳиҰҒпјү
6. вң… ж•°жҚ®з»ҹи®ЎпјҲи®ҝй—®йҮҸгҖҒж–Үз« з»ҹи®Ўпјү

```sql
    id BIGSERIAL PRIMARY KEY COMMENT 'з”ЁжҲ·ID',
    username VARCHAR(50) NOT NULL UNIQUE COMMENT 'з”ЁжҲ·еҗҚ',
    password VARCHAR(100) NOT NULL COMMENT 'еҜҶз ҒпјҲBCryptеҠ еҜҶпјү',

    avatar VARCHAR(255) COMMENT 'еӨҙеғҸURL',

    login_time TIMESTAMP COMMENT 'жңҖеҗҺзҷ»еҪ•ж—¶й—ҙ',
    create_time TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'еҲӣе»әж—¶й—ҙ',
    update_time TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'жӣҙж–°ж—¶й—ҙ',
    deleted BOOLEAN NOT NULL DEFAULT FALSE COMMENT 'йҖ»иҫ‘еҲ йҷӨ'
);

COMMENT ON TABLE sys_user IS 'з”ЁжҲ·иЎЁпјҲд»…з®ЎзҗҶе‘ҳпјү';

-- зҙўеј•
CREATE INDEX idx_sys_user_username ON sys_user(username);
CREATE INDEX idx_sys_user_deleted ON sys_user(deleted);

-- ============================================
-- з”ЁжҲ·зҷ»еҪ•ж—Ҙеҝ—иЎЁ (sys_login_log)
-- ============================================
CREATE TABLE sys_login_log (
    id BIGSERIAL PRIMARY KEY COMMENT 'ж—Ҙеҝ—ID',
    user_id BIGINT COMMENT 'з”ЁжҲ·ID',
    username VARCHAR(50) COMMENT 'з”ЁжҲ·еҗҚ',
    ip VARCHAR(50) COMMENT 'зҷ»еҪ•IP',
    location VARCHAR(100) COMMENT 'зҷ»еҪ•ең°зӮ№',
    user_agent VARCHAR(500) COMMENT 'жөҸи§ҲеҷЁUA',
    status INT NOT NULL DEFAULT 1 COMMENT 'зҠ¶жҖҒпјҡ1-жҲҗеҠҹ 0-еӨұиҙҘ',
    message VARCHAR(200) COMMENT 'ж¶ҲжҒҜ',
    create_time TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'еҲӣе»әж—¶й—ҙ'
);

COMMENT ON TABLE sys_login_log IS 'з”ЁжҲ·зҷ»еҪ•ж—Ҙеҝ—иЎЁ';

CREATE INDEX idx_login_log_user_id ON sys_login_log(user_id);
```

### 6.2 AIжңҚеҠЎе®һзҺ°

```java
@Service
@Slf4j
public class AiServiceImpl implements AiService {
    
    @Autowired
    private ChatClient chatClient;
    
    @Autowired
    private ArticleVectorMapper articleVectorMapper;
    
    /**
     * з”ҹжҲҗж–Үз« ж Үзӯҫ
     */
    @Override
    @Async("taskExecutor")
    public CompletableFuture<List<String>> generateTags(Long articleId, String content) {
        try {
            String prompt = String.format(
                "иҜ·дёәд»ҘдёӢж–Үз« з”ҹжҲҗ5-8дёӘж ҮзӯҫпјҢз”ЁйҖ—еҸ·еҲҶйҡ”гҖӮж Үзӯҫеә”иҜҘз®ҖжҙҒгҖҒзӣёе…ігҖҒзғӯй—ЁгҖӮ\n\nж–Үз« еҶ…е®№пјҲеүҚ500еӯ—пјүпјҡ\n%s",
                content.substring(0, Math.min(content.length(), 500))
            );
            
            String response = chatClient.prompt(prompt)
                .options(DashScopeChatOptions.builder()
                    .withModel("qwen-turbo")
                    .withTemperature(0.3f)
                    .build())
                .call()
                .content();
            
            List<String> tags = Arrays.stream(response.split("пјҢ|,"))
                .map(String::trim)
                .filter(s -> !s.isEmpty())
                .collect(Collectors.toList());
            
            // дҝқеӯҳз”ҹжҲҗи®°еҪ•
            saveGeneration(articleId, "tags", prompt, response);
            
            return CompletableFuture.completedFuture(tags);
        } catch (Exception e) {
            log.error("з”ҹжҲҗж ҮзӯҫеӨұиҙҘ", e);
            return CompletableFuture.completedFuture(Collections.emptyList());
        }
    }
    
    /**
     * з”ҹжҲҗж–Үз« ж‘ҳиҰҒ
     */
    @Override
    @Async("taskExecutor")
    public CompletableFuture<String> generateSummary(Long articleId, String content) {
        try {
            String prompt = String.format(
                "иҜ·дёәд»ҘдёӢж–Үз« з”ҹжҲҗ200еӯ—д»ҘеҶ…зҡ„ж‘ҳиҰҒпјҢиҰҒжұӮз®ҖжҙҒжҳҺдәҶпјҢзӘҒеҮәйҮҚзӮ№гҖӮ\n\nж–Үз« еҶ…е®№пјҲеүҚ1000еӯ—пјүпјҡ\n%s",
                content.substring(0, Math.min(content.length(), 1000))
            );
            
            String summary = chatClient.prompt(prompt)
                .options(DashScopeChatOptions.builder()
                    .withModel("qwen-turbo")
                    .withTemperature(0.3f)
                    .build())
                .call()
                .content();
            
            // дҝқеӯҳз”ҹжҲҗи®°еҪ•
            saveGeneration(articleId, "summary", prompt, summary);
            
            return CompletableFuture.completedFuture(summary);
        } catch (Exception e) {
            log.error("з”ҹжҲҗж‘ҳиҰҒеӨұиҙҘ", e);
            return CompletableFuture.completedFuture("");
        }
    }
    
    /**
     * жҗңзҙўзӣёдјјж–Үз« 
     */
    @Override
    public List<ArticleVO> searchSimilarArticles(Long articleId, int limit) {
        ArticleVector vector = articleVectorMapper.selectByArticleId(articleId);
        if (vector == null || vector.getContentVector() == null) {
            return Collections.emptyList();
        }
        
        return articleVectorMapper.findSimilarArticles(articleId, vector.getContentVector(), limit);
    }
}
```
